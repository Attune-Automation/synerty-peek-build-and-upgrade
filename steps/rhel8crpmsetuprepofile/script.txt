echo "Disabling repo management"
subscription-manager config --rhsm.manage_repos=0

echo "Removing other repositories"
rm -f /etc/yum.repos.d/*.repo

echo "Setting up rhel-8-server-rpms repository"
cat > /etc/yum.repos.d/mirror.repo << EOF
[codeready-builder]
name=OS Codeready Builder Mirror
baseurl={rpmMirrorCodereadyBuilder}
enabled=1
gpgcheck=0

[os]
name=OS RPM Mirror
baseurl={rpmMirrorBaseos}
enabled=1
gpgcheck=0

[extras]
name=Extras RPM Mirror
baseurl={rpmMirrorExtras}
enabled=1
gpgcheck=0

[appstream]
name=AppStream RPM Mirror
baseurl={rpmMirrorAppstream}
enabled=1
gpgcheck=0
EOF
